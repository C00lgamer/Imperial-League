	id = nato.1
	title = nato.1.t
	desc = nato.1.d

	is_triggered_only = yes
	
	option = {	#We accept!
		name = nato.1.o1
		
		set_country_flag = has_joined_NATO_by_event
		
		USA = {
			add_to_faction = FROM
			
			country_event = {
				id = nato.2
			}
		}
		
		ai_chance = {
			factor = 90
		}
	}
	
	option = {	#We refuse
		name = nato.1.o2
		add_political_power = 50
		USA = {
			add_opinion_modifier = {
				target = FROM
				modifier = recent_actions_negative
			}
			
			country_event = {
				id = nato.3
			}
		}
		
		ai_chance = {
			factor = 10
			
			modifier = {
				factor = 0
				AND = {
					is_historical_focus_on = yes

						tag = EST
						tag = LAT
						tag = LIT
						tag = SLO
						tag = SLV
						tag = ROM
						tag = BUL
						tag = ALB
						tag = CRO
					}
				}
			}
		}
	}
}
