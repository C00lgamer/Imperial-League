add_namespace = european_union

#Add the EU Modifiers
country_event = {
	id = european_union.1
	hidden = yes
	
	is_triggered_only = yes
	
	immediate = {
		set_country_flag = has_EU_modifiers
		every_country = {
			limit = {
				OR = {
					has_country_flag = european_union_member_state
					has_idea = idea_eu_member
				}
			}
			add_opinion_modifier = { target = FROM 	modifier = european_union_member }
			add_opinion_modifier = { target = FROM 	modifier = european_union_member_trade }
		}
	}
	
	option = {
		name = hidden_effect.option
	}
}

#Join the EU
country_event = {
	id = european_union.2
	hidden = yes
	
	is_triggered_only = yes
	
	immediate = {
		set_country_flag = european_union_member_state
		add_ideas = idea_eu_member
	}
	
	option = {
		name = hidden_effect.option
	}
}
